# Vorhersage der Preise fr die Testdaten X_test

cols = X_train.columns
importances = reg_rf.feature_importances_
std         = np.std([tree.feature_importances_ for tree in reg_rf.estimators_], axis=0)

# Subset der Trainings- und Testdaten erstellen (nur E-Bike Ja/Nein)
X_train_subset = X_train_class[y_train_class['is_ebike'].isin([0, 1])]
y_train_subset = y_train_class[y_train_class['is_ebike'].isin([0, 1])]
X_test_subset = X_test_class[y_test_class['is_ebike'].isin([0, 1])]
y_test_subset = y_test_class[y_test_class['is_ebike'].isin([0, 1])]

# Umformen der y_train and y_test zu 1-dimensional arrays
y_train_class = y_train_class.values.ravel()
y_test_class = y_test_class.values.ravel()

# Erstellen/Initialisieren des 'random forest classifier' Objekts
rfc = RandomForestClassifier(n_estimators=500, random_state=42, max_depth=5)

# Trainieren des Random Forest Classification Modells
